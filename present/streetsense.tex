\frametitle{Why Learn About the Public Infrastructure?}
\pause
\only<2->{\Large{Worse the quality of the public infrastructure, generally, the worse the quality of life.}}
	\begin{large_enum}
  	\item[-]<3-> Potholed roads $\leadsto$ slower speeds, bumpier ride
	  \item[-]<4-> Unpaved sidewalks $\leadsto$ physically disabled have tougher time
  	\item[-]<5-> Garbage $\leadsto$ foul smells, greater risk of disease
	\end{large_enum}
}

\frame{

\Large{Learning About The Quality of Streets}
\pause

\begin{large_enum}
	\item[-]<2>Open Street Maps (OSM) for data on roads
	\item[-]<3>Chunk each road into .5 km segments
	\item[-]<4>Randomly sample from the segments
	\item[-]<5>Google Street View API for lat/long of top-left of the segment with fixed direction of camera
	\item[-]<6>Concern: Google Streetview doesn't have adequate coverage. Dhaka --- 24.6\%
	\end{large_enum}
}
